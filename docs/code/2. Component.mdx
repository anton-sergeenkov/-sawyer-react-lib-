# Component

## Basic Component with Theme

> UiComponent.tsx

```tsx
import cn from "classnames"
import { checkTheme } from "../../utils"
import styles from "./styles.module.css"

export const options = {
  theme: ["light", "dark"],
}

type TTheme = "light" | "dark"

interface IProps {
  theme?: TTheme
  className?: string
  children: React.ReactNode
}

const UiComponent: React.FC<IProps> = (props) => {
  const { theme, className, children } = props

  const stylesTheme = checkTheme(theme, styles)

  return (
    <div
      className={cn(className, styles.wrapper, stylesTheme)}
    >
      Content
    </div>
  )
}
```

> styles.module.css

```css
.wrapper {
  background-color: #fff;
}
```

> index.ts

```ts
import { options } from "./UiComponent"

export { default } from "./UiComponent"
export { options }
```
